Результат
Форма Бэкуса-Наура
Язык = "Программа" Звенья "Конец"
Звенья = Звено ";" ... Звено
Звено = "Ввод текста" Оператор ";" ... Оператор
Оператор = Метка ... Метка ":" Переменная "=" Правая часть
Правая часть = Блок Знак1 ... Блок
Знак1 = "+" ! "-"
Блок = Часть Знак2 ... Часть
Знак2 = "*" ! "/"
Часть = Кусок "^" ... Кусок
Кусок = Переменная ! Целое ! "(" Правая часть ")" ! "[" Правая часть "]"
/// Глубина вложенности [ ] до 2 ///
Переменная = Буква Цифра Цифра Цифра
Метка = Целое
Целое = Цифра ... Цифра
Буква = "А" ! "Б" ! "В" ! ... ! "Я"
Цифра = "0" ! "1" ! "2" ! ... ! "7"
